238  int drawSystemInfos(DebugSystems systems, SystemInterfaceFlags type)
239  {
240  SystemInfo[] systemInfos = null;
241 
242  switch (type)
243  {
244  case SystemInterfaceFlags.IInitializeSystem:
245  systemInfos = systems.initializeSystemInfos
246  .Where(systemInfo => systemInfo.initializationDuration >= _threshold)
247  .ToArray();
248  break;
249  case SystemInterfaceFlags.IExecuteSystem:
250  systemInfos = systems.executeSystemInfos
251  .Where(systemInfo => systemInfo.averageExecutionDuration >= _threshold)
252  .ToArray();
253  break;
254  case SystemInterfaceFlags.ICleanupSystem:
255  systemInfos = systems.cleanupSystemInfos
256  .Where(systemInfo => systemInfo.cleanupDuration >= _threshold)
257  .ToArray();
258  break;
259  case SystemInterfaceFlags.ITearDownSystem:
260  systemInfos = systems.tearDownSystemInfos
261  .Where(systemInfo => systemInfo.teardownDuration >= _threshold)
262  .ToArray();
263  break;
264  }
265 
266  systemInfos = getSortedSystemInfos(systemInfos, _systemSortMethod);
267 
268  var systemsDrawn = 0;
269  foreach (var systemInfo in systemInfos)
270  {
271  var debugSystems = systemInfo.system as DebugSystems;
272  if (debugSystems != null)
273  {
274  if (!shouldShowSystems(debugSystems, type))
275  {
276  continue;
277  }
278  }
279 
280  if (EditorLayout.MatchesSearchString(systemInfo.systemName.ToLower(), _systemNameSearchString.ToLower()))
281  {
282  EditorGUILayout.BeginHorizontal();
283  {
284  var indent = EditorGUI.indentLevel;
285  EditorGUI.indentLevel = 0;
286 
287  var wasActive = systemInfo.isActive;
288  if (systemInfo.areAllParentsActive)
289  {
290  systemInfo.isActive = EditorGUILayout.Toggle(systemInfo.isActive, GUILayout.Width(20));
291  }
292  else
293  {
294  EditorGUI.BeginDisabledGroup(true);
295  {
296  EditorGUILayout.Toggle(false, GUILayout.Width(20));
297  }
298  }
299  EditorGUI.EndDisabledGroup();
300 
301  EditorGUI.indentLevel = indent;
302 
303  if (systemInfo.isActive != wasActive)
304  {
305  var reactiveSystem = systemInfo.system as IReactiveSystem;
306  if (reactiveSystem != null)
307  {
308  if (systemInfo.isActive)
309  {
310  reactiveSystem.Activate();
311  }
312  else
313  {
314  reactiveSystem.Deactivate();
315  }
316  }
317  }
318 
319  switch (type)
320  {
321  case SystemInterfaceFlags.IInitializeSystem:
322  EditorGUILayout.LabelField(systemInfo.systemName, systemInfo.initializationDuration.ToString(), getSystemStyle(systemInfo, SystemInterfaceFlags.IInitializeSystem));
323  break;
324  case SystemInterfaceFlags.IExecuteSystem:
325  var avgE = string.Format("Ø {0:00.000}", systemInfo.averageExecutionDuration).PadRight(12);
326  var minE = string.Format("▼ {0:00.000}", systemInfo.minExecutionDuration).PadRight(12);
327  var maxE = string.Format("▲ {0:00.000}", systemInfo.maxExecutionDuration);
328  EditorGUILayout.LabelField(systemInfo.systemName, avgE + minE + maxE, getSystemStyle(systemInfo, SystemInterfaceFlags.IExecuteSystem));
329  break;
330  case SystemInterfaceFlags.ICleanupSystem:
331  var avgC = string.Format("Ø {0:00.000}", systemInfo.averageCleanupDuration).PadRight(12);
332  var minC = string.Format("▼ {0:00.000}", systemInfo.minCleanupDuration).PadRight(12);
333  var maxC = string.Format("▲ {0:00.000}", systemInfo.maxCleanupDuration);
334  EditorGUILayout.LabelField(systemInfo.systemName, avgC + minC + maxC, getSystemStyle(systemInfo, SystemInterfaceFlags.ICleanupSystem));
335  break;
336  case SystemInterfaceFlags.ITearDownSystem:
337  EditorGUILayout.LabelField(systemInfo.systemName, systemInfo.teardownDuration.ToString(), getSystemStyle(systemInfo, SystemInterfaceFlags.ITearDownSystem));
338  break;
339  }
340  }
341  EditorGUILayout.EndHorizontal();
342 
343  systemsDrawn += 1;
344  }
345 
346  var debugSystem = systemInfo.system as DebugSystems;
347  if (debugSystem != null)
348  {
349  var indent = EditorGUI.indentLevel;
350  EditorGUI.indentLevel += 1;
351  systemsDrawn += drawSystemInfos(debugSystem, type);
352  EditorGUI.indentLevel = indent;
353  }
354  }
355 
356  return systemsDrawn;

359  static SystemInfo[] getSortedSystemInfos(SystemInfo[] systemInfos, SortMethod sortMethod)
360  {
361  if (sortMethod == SortMethod.Name)
362  {
363  return systemInfos
364  .OrderBy(systemInfo => systemInfo.systemName)
365  .ToArray();
366  }
367  if (sortMethod == SortMethod.NameDescending)
368  {
369  return systemInfos
370  .OrderByDescending(systemInfo => systemInfo.systemName)
371  .ToArray();
372  }
373 
374  if (sortMethod == SortMethod.ExecutionTime)
375  {
376  return systemInfos
377  .OrderBy(systemInfo => systemInfo.averageExecutionDuration)
378  .ToArray();
379  }
380  if (sortMethod == SortMethod.ExecutionTimeDescending)
381  {
382  return systemInfos
383  .OrderByDescending(systemInfo => systemInfo.averageExecutionDuration)
384  .ToArray();
385  }
386 
387  return systemInfos;

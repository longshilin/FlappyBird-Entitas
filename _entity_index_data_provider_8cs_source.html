50  public CodeGeneratorData[] GetData() {

52  .GetCachedAssemblyResolver(objectCache, _assembliesConfig.assemblies, _codeGeneratorConfig.searchPaths)
53  .GetTypes();
54 
55  var entityIndexData = types
56  .Where(type => !type.IsAbstract)
57  .Where(type => type.ImplementsInterface<IComponent>())
58  .ToDictionary(
59  type => type,
60  type => type.GetPublicMemberInfos())
61  .Where(kv => kv.Value.Any(info => info.attributes.Any(attr => attr.attribute is AbstractEntityIndexAttribute)))
62  .SelectMany(kv => createEntityIndexData(kv.Key, kv.Value));
63 
64  var customEntityIndexData = types
65  .Where(type => !type.IsAbstract)
66  .Where(type => Attribute.IsDefined(type, typeof(CustomEntityIndexAttribute)))
67  .Select(createCustomEntityIndexData);

70  .Concat(customEntityIndexData)

74  EntityIndexData[] createEntityIndexData(Type type, List<PublicMemberInfo> infos) {
75  var hasMultiple = infos.Count(i => i.attributes.Count(attr => attr.attribute is AbstractEntityIndexAttribute) == 1) > 1;
76  return infos
77  .Where(i => i.attributes.Count(attr => attr.attribute is AbstractEntityIndexAttribute) == 1)
78  .Select(info => {
79  var data = new EntityIndexData();
80  var attribute = (AbstractEntityIndexAttribute)info.attributes.Single(attr => attr.attribute is AbstractEntityIndexAttribute).attribute;
81 
82  data.SetEntityIndexType(getEntityIndexType(attribute));
83  data.IsCustom(false);
84  data.SetEntityIndexName(type.ToCompilableString().ToComponentName(_ignoreNamespacesConfig.ignoreNamespaces));
85  data.SetKeyType(info.type.ToCompilableString());
86  data.SetComponentType(type.ToCompilableString());
87  data.SetMemberName(info.name);
88  data.SetHasMultiple(hasMultiple);
89  data.SetContextNames(_contextsComponentDataProvider.GetContextNamesOrDefault(type));

91  return data;

95  EntityIndexData createCustomEntityIndexData(Type type) {

98  var attribute = (CustomEntityIndexAttribute)type.GetCustomAttributes(typeof(CustomEntityIndexAttribute), false)[0];
99 
100  data.SetEntityIndexType(type.ToCompilableString());
101  data.IsCustom(true);
102  data.SetEntityIndexName(type.ToCompilableString().RemoveDots());
103  data.SetHasMultiple(false);
104  data.SetContextNames(new[] { attribute.contextType.ToCompilableString().ShortTypeName().RemoveContextSuffix() });
105 
106  var getMethods = type
107  .GetMethods(BindingFlags.Public | BindingFlags.Instance)
108  .Where(method => Attribute.IsDefined(method, typeof(EntityIndexGetMethodAttribute)))
109  .Select(method => new MethodData(
110  method.ReturnType.ToCompilableString(),
111  method.Name,
112  method.GetParameters()
113  .Select(p => new MemberData(p.ParameterType.ToCompilableString(), p.Name))
114  .ToArray()
115  )).ToArray();
116 
117  data.SetCustomMethods(getMethods);
118 
119  return data;

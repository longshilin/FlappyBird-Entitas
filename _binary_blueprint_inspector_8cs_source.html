46  public static bool UpdateBinaryBlueprint(BinaryBlueprint binaryBlueprint, IContext[] allContexts, string[] allContextNames) {
47  var blueprint = binaryBlueprint.Deserialize();
48  var needsUpdate = false;
49 
50  var contextIndex = Array.IndexOf(allContextNames, blueprint.contextIdentifier);
51  if (contextIndex < 0) {
52  contextIndex = 0;
53  needsUpdate = true;
54  }
55 
56  var context = allContexts[contextIndex];
57  blueprint.contextIdentifier = context.contextInfo.name;
58 
59  foreach (var component in blueprint.components) {
60  var type = component.fullTypeName.ToType();
61  var index = Array.IndexOf(context.contextInfo.componentTypes, type);
62 
63  if (index != component.index) {
64  Debug.Log(string.Format(
65  "Blueprint '{0}' has invalid or outdated component index for '{1}'. Index was {2} but should be {3}. Updated index.",
66  blueprint.name, component.fullTypeName, component.index, index));
67 
68  component.index = index;
69  needsUpdate = true;
70  }
71  }
72 
73  if (needsUpdate) {
74  Debug.Log("Updating Blueprint '" + blueprint.name + "'");
75  binaryBlueprint.Serialize(blueprint);
76  }
77 
78  return needsUpdate;
79  }
